144 struct TexParams

148 
149  TexParams() {}
150  TexParams(TexFilter f, WrapMode w) : filter(f), wrap(w,w) {}
151  TexParams(TexFilter f, WrapMode u, WrapMode v) : filter(f), wrap(u,v) {}
152 
153  static TexParams clamped(TexFilter f=TexFilter::DontCare) { return TexParams(f, WrapMode::Clamp); }
154  static TexParams repeated(TexFilter f=TexFilter::DontCare) { return TexParams(f, WrapMode::Repeat); }
155  static TexParams clamped_nearest() { return clamped(TexFilter::Nearest); }
156  static TexParams clamped_linear() { return clamped(TexFilter::Linear); }
157  static TexParams clamped_mipmapped() { return clamped(TexFilter::Mipmapped); }
158  static TexParams repeated_linear() { return repeated(TexFilter::Linear); }
159  static TexParams repeated_mipmapped() { return repeated(TexFilter::Mipmapped); }
160  static TexParams mipmapped(WrapMode u, WrapMode v) { return TexParams(TexFilter::Mipmapped, u, v); }
161 
162  friend bool operator<(const TexParams& a, const TexParams& b)
163  {
164  if (a.filter != b.filter) { return a.filter < b.filter; }
165  if (a.wrap != b.wrap) { return a.wrap < b.wrap; }
166  return false; // Same
167  }

170  {
171  return a.filter == b.filter && a.wrap == b.wrap;
