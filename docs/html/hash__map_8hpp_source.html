520  // Find the bucket with this key, or return nullptr
521  size_t find_filled_bucket(const KeyT& key) const
522  {
523  if (empty()) { return (size_t)-1; } // Optimization
524 
525  auto hash_value = _hasher(key);
526  for (int offset=0; offset<=_max_probe_length; ++offset) {
527  auto bucket = (hash_value + offset) & _mask;
528  if (_states[bucket] == State::FILLED && _comp(_pairs[bucket].first, key)) {
529  return bucket;
530  }
531  if (_states[bucket] == State::INACTIVE) {
532  return (size_t)-1; // End of the chain!
533  }
534  }
535  return (size_t)-1;
536  }

538  // Find the bucket with this key, or return a good empty bucket to place the key in.
539  // In the latter case, the bucket is expected to be filled.
540  size_t find_or_allocate(const KeyT& key)
541  {
542  auto hash_value = _hasher(key);
543  size_t hole = (size_t)-1;
544  int offset=0;
545  for (; offset<=_max_probe_length; ++offset) {
546  auto bucket = (hash_value + offset) & _mask;
547 
548  if (_states[bucket] == State::FILLED) {
549  if (_comp(_pairs[bucket].first, key)) {
550  return bucket;
551  }
552  } else if (_states[bucket] == State::INACTIVE) {
553  return bucket;
554  } else {
555  // ACTIVE: keep searching
556  if (hole == (size_t)-1) {
557  hole = bucket;
558  }
559  }
560  }
561 
562  // No key found - but maybe a hole for it
563 
564  DCHECK_EQ_F(offset, _max_probe_length+1);
565 
566  if (hole != (size_t)-1) {
567  return hole;
568  }
569 
570  // No hole found within _max_probe_length
571  for (; ; ++offset) {
572  auto bucket = (hash_value + offset) & _mask;
573 
574  if (_states[bucket] != State::FILLED) {
575  _max_probe_length = offset;
576  return bucket;
577  }
578  }
579  }
580 
581  // key is not in this map. Find a place to put it.
582  size_t find_empty_bucket(const KeyT& key)
583  {
584  auto hash_value = _hasher(key);
585  for (int offset=0; ; ++offset) {
586  auto bucket = (hash_value + offset) & _mask;
587  if (_states[bucket] != State::FILLED) {
588  if (offset > _max_probe_length) {
589  _max_probe_length = offset;
590  }
591  return bucket;
592  }
