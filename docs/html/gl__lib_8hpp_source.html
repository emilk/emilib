429 
430  void add_strip(const Vertex* ptr, size_t n)
431  {
432  if (n == 0) { return; }
433  CHECK_GT_F(n, 2u);
434  VBO& vbo = _mesh_painter.vert_vbo();
435  if (!vbo.empty()) {
436  // Connector by degenerate triangles:
437  Vertex last_vertex = vbo.data<Vertex>()[vbo.count() - 1u];
438  vbo.append<Vertex>(&last_vertex, 1u);
439  vbo.append<Vertex>(ptr, 1u);
440  }
441  vbo.append<Vertex>(ptr, n);
442  }
443
